// API endpoint untuk upload SMS app.post('/api/upload-sms', async (req, res) => { const { sender, message, date, time, format } = req.body; try { // Generate SMS image const imageBuffer = await generateSMSImage({ sender, message, date, time, format }); // Upload ke multiple CDN const results = await Promise.allSettled([ uploadToTmpFiles(imageBuffer), uploadToFileIO(imageBuffer), uploadToTelegraph(imageBuffer) ]); const cdnUrls = {}; results.forEach((result, index) => { if (result.status === 'fulfilled') { cdnUrls[['tmpfiles', 'fileio', 'telegraph'][index]] = result.value; } }); res.json({ success: true, data: { cdnUrls, fileSize: formatBytes(imageBuffer.length), message: `Upload berhasil ke ${Object.keys(cdnUrls).length} server` } }); } catch (error) { res.status(500).json({ success: false, error: error.message }); } });
